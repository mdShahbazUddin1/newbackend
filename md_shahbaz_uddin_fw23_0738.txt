// What is a closure ?

-->In closure .inner Function has accessbility to outer scope


function counter(){

let count = 0;

// in inner function count is accessible to its parent function//
function incCounter(){
count++;
return count 

}

return incCounter

}

let ans = counter();

ans()// here we will get 1
ans()// here we will get 2 because we are just return incCounter to its parent 
and the previous value was 1 and current value will be 2
